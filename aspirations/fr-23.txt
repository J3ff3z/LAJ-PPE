<script>
document.addEventListener("DOMContentLoaded", function () {

    // --------------------------------------------
    // Helper: read width ignoring transform:scale
    // --------------------------------------------
    function getUnscaledWidth(el) {
        const cs = window.getComputedStyle(el);
        const transform = cs.transform;

        if (transform && transform !== "none") {
            const old = el.style.transform;
            el.style.transform = "none";
            const w = el.getBoundingClientRect().width;
            el.style.transform = old;
            return w;
        }

        return el.getBoundingClientRect().width;
    }

    // --------------------------------------------
    // Compute visible bitmap (object-fit: contain)
    // --------------------------------------------
    function getVisibleSize(img) {
        const box = img.getBoundingClientRect();
        const boxW = box.width;
        const boxH = box.height;

        const natW = img.naturalWidth;
        const natH = img.naturalHeight;

        if (!natW || !natH || !boxW || !boxH) {
            return { w: boxW, h: boxH };
        }

        const imgRatio = natW / natH;
        const boxRatio = boxW / boxH;

        let visibleW, visibleH;

        if (imgRatio > boxRatio) {
            visibleW = boxW;
            visibleH = boxW / imgRatio;
        } else {
            visibleH = boxH;
            visibleW = boxH * imgRatio;
        }

        return { w: visibleW, h: visibleH };
    }

    // --------------------------------------------
    // Main logic
    // --------------------------------------------
    function syncCaptionWidths() {
        const figures = document.querySelectorAll("figure");

        figures.forEach(figure => {
            const img = figure.querySelector("img");
            const caption = figure.querySelector("figcaption");
            if (!img || !caption) return;

            function apply() {
                const { w: visibleW } = getVisibleSize(img);

                const figureRect = figure.getBoundingClientRect();
                const containerW = figureRect.width;

                const finalWidth = Math.min(visibleW + 5, containerW);
                const safeWidth = Math.min(finalWidth, visibleW);

                caption.style.setProperty("width", safeWidth + "px", "important");
                caption.style.setProperty("position", "relative", "important");

                const captionH = caption.offsetHeight;
                caption.style.setProperty("margin-top", (-captionH) + "px", "important");

                const imgRect = img.getBoundingClientRect();
                const style = window.getComputedStyle(img);
                const { w: bmpW } = getVisibleSize(img);

                let posX = style.objectPosition.split(" ")[0].trim().toLowerCase();

                let bmpLeft = imgRect.left;

                if (posX === "left" || posX === "0%" || posX === "0px") {
                    bmpLeft += 0;
                } else if (posX === "right" || posX === "100%") {
                    bmpLeft += (imgRect.width - bmpW);
                } else {
                    bmpLeft += (imgRect.width - bmpW) / 2;
                }

                const offsetX = bmpLeft - figureRect.left;

                caption.style.setProperty("margin-left", offsetX + "px", "important");
            }

            function applyStable() {
                apply();
                requestAnimationFrame(apply);
                setTimeout(apply, 30);
                setTimeout(apply, 120);
                setTimeout(apply, 250);
            }

            function startStabilizer() {
                let lastWidth = 0;
                let stableCount = 0;

                function loop() {
                    const now = getUnscaledWidth(figure);

                    if (Math.abs(now - lastWidth) < 0.5) {
                        stableCount++;
                    } else {
                        stableCount = 0;
                        apply();
                    }

                    lastWidth = now;

                    if (stableCount < 4) {
                        requestAnimationFrame(loop);
                    } else {
                        apply();
                    }
                }

                loop();
            }

            function initialize() {
                applyStable();
                startStabilizer();
            }

            // --------------------------------------------
            // Decode first for true intrinsic size
            // --------------------------------------------
            if (img.decode) {
                img.decode().then(initialize).catch(initialize);
            } else {
                if (img.complete) initialize();
                else img.addEventListener("load", initialize);
            }

            // --------------------------------------------
            // Handle responsive srcset changes after init
            // --------------------------------------------
            img.addEventListener("load", () => {
                applyStable();
                startStabilizer();
            });

            // --------------------------------------------
            // Resize observer for true size changes
            // --------------------------------------------
            const ro = new ResizeObserver(applyStable);
            ro.observe(img);
            ro.observe(figure);

            // --------------------------------------------
            // attribute observer for src/srcset/sizes/style
            // --------------------------------------------
            const mo = new MutationObserver(applyStable);
            mo.observe(img, {
                attributes: true,
                attributeFilter: ["src", "srcset", "sizes", "style"]
            });

            window.addEventListener("resize", apply);
        });
    }

    syncCaptionWidths();

    // --------------------------------------------
    // Handle webfont loading (delayed layout change)
    // --------------------------------------------
    if (document.fonts && document.fonts.ready) {
        document.fonts.ready.then(() => {
            syncCaptionWidths();
        });
    }

});

// --------------------------------------------
// Handle back/forward cache restore
// --------------------------------------------
window.addEventListener("pageshow", function (event) {
    if (event.persisted) {
        syncCaptionWidths();
    }
});</script>

<div class="wp-block-group has-global-padding is-layout-constrained wp-block-group-is-layout-constrained"><h1 style="letter-spacing:-0.4px; margin-top:0;margin-bottom:0;" class="wp-block-post-title has-bodoni-moda-font-family"><a href="https://amavero.fr/farrokhzad-forough-le-peche/" target="_self" >Farrokhzad (Forough) : Le péché</a></h1></div>



<div class="wp-block-group alignfull is-layout-flow wp-block-group-is-layout-flow" style="padding-top:0;padding-bottom:0">
<div class="wp-block-group alignfull is-layout-flow wp-block-group-is-layout-flow">
<div class="wp-block-group alignfull is-layout-flow wp-block-group-is-layout-flow"><div class="entry-content alignfull wp-block-post-content has-bricolage-grotesque-font-family has-global-padding is-layout-constrained wp-block-post-content-is-layout-constrained">
<div class="wp-block-columns is-layout-flex wp-container-core-columns-is-layout-28f84493 wp-block-columns-is-layout-flex">
<div class="wp-block-column is-layout-flow wp-block-column-is-layout-flow" style="flex-basis:100%">
<div class="wp-block-group has-global-padding is-layout-constrained wp-block-group-is-layout-constrained">
<div class="wp-block-columns is-layout-flex wp-container-core-columns-is-layout-28f84493 wp-block-columns-is-layout-flex">
<div class="wp-block-column is-layout-flow wp-block-column-is-layout-flow" style="flex-basis:80%">
<p>J&rsquo;ai pêché, pêché dans le plaisir,<br>dans des bras chauds et enflammés,<br>j&rsquo;ai pêché dans des bras de fer,<br>brûlants et rancuniers.</p>



<p>Dans ce lieu solitaire, sombre et muet,<br>ses yeux remplis de mystère j&rsquo;ai regardé,<br>mon coeur dans ma poitrine, impatiemment a tremblé,<br>des supplications de désirs de ses yeux.</p>



<p>Dans ce lieu solitaire, sombre et muet,<br>je me suis assise près de lui, agitée,<br>sa lèvre, l&rsquo;envie, sur mes lèvres a versée,<br>de la tristesse de mon coeur fou, je me suis libérée.</p>



<p>A l&rsquo;oreille, l&rsquo;histoire d&rsquo;amour, je lui ai racontée,<br>je te veux mon amant,<br>je te veux, toi dont les bras sont vivifiants,<br>je te veux, toi mon amoureux fou.</p>



<p>Le désir alluma le feu dans son regard,<br>le vin rouge dansa dans le verre,<br>mon corps sur le lit doux,<br>dans l&rsquo;ivresse trembla sur sa poitrine.</p>



<p>J&rsquo;ai pêché dans le plaisir,<br>près d&rsquo;un corps tremblant et évanoui,<br>Dieu! Je ne sais ce que j&rsquo;ai fait,<br>dans ce lieu solitaire, sombre et muet&#8230;</p>

// Recherche et filtres pour l'index global uniquement
document.addEventListener('DOMContentLoaded', function() {
    console.log('=== Init recherche et filtres ===');
    
    // Vérifier qu'on est sur une page avec l'index global
    const indexGlobal = document.querySelector('.amv-index-global');
    if (!indexGlobal) {
        console.log('Pas d\'index global sur cette page');
        return;
    }
    
    const searchInput = document.getElementById('amv-search-input');
    const searchClear = document.getElementById('amv-search-clear');
    const filterBtns = document.querySelectorAll('.amv-filter-btn');
    const auteurItems = document.querySelectorAll('.amv-auteur-item');
    const lettreHeaders = document.querySelectorAll('.amv-lettre-header');
    const alphaNav = document.getElementById('amv-alpha-nav-global');
    const noResults = document.querySelector('.amv-no-results');
    
    let currentFilter = 'all';
    let currentSearch = '';
    
    // Fonction de filtrage
    function applyFilters() {
        let visibleCount = 0;
        const visibleLetters = new Set();
        
        auteurItems.forEach(function(item) {
            const authorName = item.getAttribute('data-author') || '';
            const categories = item.getAttribute('data-categories') || '';
            
           	// Vérifier le filtre de catégorie avec correspondance exacte
			let categoryMatch = false;
			if (currentFilter === 'all') {
    		categoryMatch = true;
			} else {
    		// Split les catégories et vérifier la correspondance exacte
    		const catArray = categories.split(',');
    		categoryMatch = catArray.includes(currentFilter);
			}
            
            // Vérifier la recherche
            let searchMatch = currentSearch === '' || authorName.toLowerCase().includes(currentSearch.toLowerCase());
            
            // Afficher/masquer l'item
            if (categoryMatch && searchMatch) {
                item.classList.remove('hidden');
                visibleCount++;
                
                // Récupérer la lettre de cet auteur
                const lettre = item.closest('.amv-auteurs-grid')?.previousElementSibling?.getAttribute('data-letter');
                if (lettre) {
                    visibleLetters.add(lettre);
                }
                
                // Recalculer le nombre affiché dans le nom selon le filtre
                const nomElement = item.querySelector('.amv-auteur-nom');
                if (nomElement) {
                    if (currentFilter !== 'all') {
                        // Compter seulement les publications de la catégorie filtrée
                        const categoryGroups = item.querySelectorAll('.amv-category-group[data-category="' + currentFilter + '"]');
                        let countForCategory = 0;
                        
                        categoryGroups.forEach(function(group) {
                            const items = group.querySelectorAll('ul li');
                            countForCategory += items.length;
                        });
                        
                        // Mettre à jour le texte avec le nouveau nombre
                        const nomText = nomElement.textContent.replace(/\s*\(\d+\)/, '');
                        nomElement.textContent = nomText + ' (' + countForCategory + ')';
                    } else {
                        // Remettre le nombre total
                        const categoryGroups = item.querySelectorAll('.amv-category-group');
                        let totalCount = 0;
                        
                        categoryGroups.forEach(function(group) {
                            const items = group.querySelectorAll('ul li');
                            totalCount += items.length;
                        });
                        
                        const nomText = nomElement.textContent.replace(/\s*\(\d+\)/, '');
                        nomElement.textContent = nomText + ' (' + totalCount + ')';
                    }
                }
            } else {
                item.classList.add('hidden');
            }
            
            // Gérer les catégories dans l'index global
            if (currentFilter !== 'all' && categoryMatch && searchMatch) {
                const categoryGroups = item.querySelectorAll('.amv-category-group');
                categoryGroups.forEach(function(group) {
                    const groupCategory = group.getAttribute('data-category');
                    if (groupCategory === currentFilter) {
                        group.classList.remove('hidden');
                    } else {
                        group.classList.add('hidden');
                    }
                });
            } else if (currentFilter === 'all') {
                const categoryGroups = item.querySelectorAll('.amv-category-group');
                categoryGroups.forEach(function(group) {
                    group.classList.remove('hidden');
                });
            }
        });
        
        // Masquer/afficher les headers de lettres
        lettreHeaders.forEach(function(header) {
            const lettre = header.getAttribute('data-letter');
            const grid = header.nextElementSibling;
            
            if (grid && grid.classList.contains('amv-auteurs-grid')) {
                const visibleItems = grid.querySelectorAll('.amv-auteur-item:not(.hidden)');
                
                if (visibleItems.length > 0) {
                    header.classList.remove('hidden');
                } else {
                    header.classList.add('hidden');
                }
            }
        });
        
        // Mettre à jour la navigation alphabétique
        if (alphaNav) {
            const alphaLinks = alphaNav.querySelectorAll('a, span');
            alphaLinks.forEach(function(link) {
                const lettre = link.getAttribute('data-letter');
                if (visibleLetters.has(lettre)) {
                    link.classList.remove('hidden');
                } else {
                    link.classList.add('hidden');
                }
            });
        }
        
        // Message "aucun résultat"
        if (noResults) {
            if (visibleCount === 0) {
                noResults.style.display = 'block';
            } else {
                noResults.style.display = 'none';
            }
        }
    }
    
    // Gestion de la recherche
    if (searchInput) {
        searchInput.addEventListener('input', function() {
            currentSearch = this.value.trim();
            
            if (currentSearch !== '') {
                searchClear.style.display = 'block';
            } else {
                searchClear.style.display = 'none';
            }
            
            applyFilters();
        });
    }
    
    // Bouton clear recherche
    if (searchClear) {
        searchClear.addEventListener('click', function() {
            searchInput.value = '';
            currentSearch = '';
            this.style.display = 'none';
            applyFilters();
        });
    }
    
    // Gestion des filtres de catégorie
    filterBtns.forEach(function(btn) {
        btn.addEventListener('click', function() {
            // Retirer la classe active de tous les boutons
            filterBtns.forEach(function(b) {
                b.classList.remove('active');
            });
            
            // Ajouter la classe active au bouton cliqué
            this.classList.add('active');
            
            // Mettre à jour le filtre actuel
            currentFilter = this.getAttribute('data-category');
            
            // Appliquer les filtres
            applyFilters();
        });
