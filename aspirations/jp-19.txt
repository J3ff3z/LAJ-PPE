  var logoUrl = logoUrl = "/base/images/header-logo.2025012002.png";
  var defaultXPostMessage = '選択範囲の画像を「貼り付け」できます！';

document.addEventListener('DOMContentLoaded', function() {
  var initialZoomScale = 1;
  // ページの初期スケールを少し遅れて取得
  setTimeout(function() {
    if (window.visualViewport) {
      initialZoomScale = window.visualViewport.scale;
    }
  }, 100);

  // --- シェアボタンの生成 ---
  var shareButton = document.createElement('div');
  shareButton.id = 'selection-to-image-share-button';
  var shareButtonStyles = {
    display: 'none',
    position: 'absolute',
    backgroundColor: 'black',
    color: 'white',
    padding: '5px 10px',
    borderRadius: '5px',
    cursor: 'pointer',
    zIndex: '1000',
    alignItems: 'center',
    gap: '5px'
  };
  for (var styleKey in shareButtonStyles) {
    shareButton.style[styleKey] = shareButtonStyles[styleKey];
  }
  shareButton.innerHTML = '<svg viewBox="0 0 24 24" fill="#fff" width="22" aria-hidden="true" class="r-1nao33i r-4qtqp9 r-yyyyoo r-16y2uox r-8kz0gk r-dnmrzs r-bnwqim r-1plcrui r-lrvibr r-lrsllp"><g><path d="M18.244 2.25h3.308l-7.227 8.26 8.502 11.24H16.17l-5.214-6.817L4.99 21.75H1.68l7.73-8.835L1.254 2.25H8.08l4.713 6.231zm-1.161 17.52h1.833L7.084 4.126H5.117z"></path></g></svg>に投稿する';
  document.body.appendChild(shareButton);

  // --- 変数定義 ---
  var currentRange = null; // 現在選択されているテキストの範囲
  var selectionChangeTimeout = null; // selectionchangeイベントの遅延実行用タイマー

  // --- タッチデバイスかどうかの判定 ---
  function isTouchDevice() {
    return 'ontouchstart' in window || navigator.maxTouchPoints > 0;
  }

  // --- イベントリスナー ---
  // --- PC用ロジック (マウス追従) ---
  document.addEventListener('mouseup', function(e) {
    // タッチデバイスでは何もしない
    if (isTouchDevice()) return;

    // シェアボタン自身やその子要素をクリックした場合は何もしない
    if (e.target.closest('#selection-to-image-share-button')) {
      return;
    }
    // mouseup直後だと選択が解除されることがあるため、わずかに遅延させる
    setTimeout(function() {
      var selection = window.getSelection();
      if (selection && selection.toString().trim().length > 0) {
        currentRange = selection.getRangeAt(0);
        shareButton.style.left = (e.pageX + 5) + 'px';
        shareButton.style.top = (e.pageY + 5) + 'px';
        shareButton.style.display = 'flex';
      }
    }, 10);
  });

  // --- スマートフォン用ロジック (選択範囲の中央下) ---
  document.addEventListener('selectionchange', function() {
    // PCでは何もしない
    if (!isTouchDevice()) return;

    // 連続するイベント発火を抑制するため、タイマーをクリア
    clearTimeout(selectionChangeTimeout);
    // 選択操作が終わるのを待ってから処理を実行
    selectionChangeTimeout = setTimeout(function() {
      var selection = window.getSelection();
      if (selection && selection.toString().trim().length > 0) {
        currentRange = selection.getRangeAt(0);
        var rect = currentRange.getBoundingClientRect();
        if (rect.width > 0) {
          var viewport = window.visualViewport || {
            scale: 1,
            offsetLeft: window.scrollX,
            width: window.innerWidth
          };

          // --- ボタンサイズの動的調整 ---
          function getViewportWidth() {
            var viewportMeta = document.querySelector('meta[name="viewport"]');
            if (viewportMeta) {
                var content = viewportMeta.getAttribute('content');
                if (content.includes('width=device-width')) {
                  return screen.width;
                }
                var match = content.match(/width\s*=\s*([0-9]+)/);
                if (match && match[1]) {
                    return parseInt(match[1], 10);
                }
            }
            return 1040; // デフォルト値
          }

          var viewportWidth = getViewportWidth();
          var scale = screen.width / viewportWidth; // 実際の画面幅とviewport幅の比率

          // フォントサイズの調整（最小14px）
          var scaledFontSize = Math.max(16 / scale, 14);
          shareButton.style.fontSize = scaledFontSize + 'px';

          // パディングの調整
          var scaledPadding = Math.max(10 / scale, 8);
          shareButton.style.padding = scaledPadding + 'px ' + (scaledPadding * 2) + 'px';

          // SVGアイコンサイズの調整
          var icon = shareButton.querySelector('svg');
          if (icon) {
            var iconSize = Math.max(22 / scale, 18);
            icon.setAttribute('width', iconSize);
          }

          // --- ピンチアウト時の過度な拡大を抑制 ---
          var currentZoomScale = viewport.scale;
          // ユーザー操作によるズーム倍率を計算
          var userZoomFactor = currentZoomScale / initialZoomScale;
          var inverseScale = 1 / userZoomFactor;
          shareButton.style.transform = 'scale(' + inverseScale + ')';
          shareButton.style.transformOrigin = 'center bottom';

          // ボタンの幅を取得するために一度表示させる（画面外）
          shareButton.style.display = 'flex';
          shareButton.style.left = '-9999px';
          shareButton.style.top = '-9999px';

          var buttonWidth = shareButton.offsetWidth;
          var buttonHeight = shareButton.offsetHeight;

          // ピンチイン・アウトに対応するため、表示領域(visualViewport)の中心にボタンを配置
          var x = viewport.offsetLeft + (viewport.width / 2) - (buttonWidth / 2);
          var y = rect.bottom + window.scrollY;

          // --- マージンの動的調整 ---
          // transform:scaleによって生じる隙間を計算して補正する
          var visualGap = buttonHeight - (buttonHeight * inverseScale);
          var baseMargin = 25; // 基準マージン
          var scaledMargin = (baseMargin / userZoomFactor) - visualGap;

          shareButton.style.left = x + 'px';
          shareButton.style.top = (y + scaledMargin) + 'px';
        }
      } else {
        // 選択が解除されたらボタンを非表示
        shareButton.style.display = 'none';
      }
    }, 500); // 選択が終わってから500ms後に実行
  });

  // シェアボタン以外をクリックした場合はボタンを非表示にする
  function handleInteractionStart(e) {
    if (!e.target.closest('#selection-to-image-share-button')) {
      shareButton.style.display = 'none';
      currentRange = null;
    }
  }
  document.addEventListener('mousedown', handleInteractionStart);
  document.addEventListener('touchstart', handleInteractionStart);

  // --- シェアボタンクリック時の処理 ---
  shareButton.addEventListener('click', function() {
    if (!currentRange) return;

    // --- 選択範囲を画像化するための準備 ---
    var fragment = currentRange.cloneContents();
    var commonAncestor = currentRange.commonAncestorContainer;
    var style = window.getComputedStyle(commonAncestor.nodeType === 1 ? commonAncestor : commonAncestor.parentElement);
    var containerToRender, containerToRemove;
    var snapshotContainer = document.createElement('div');
    var snapshotContainerStyles = {
        fontFamily: style.fontFamily,
        fontSize: style.fontSize,
        color: style.color,
        backgroundColor: style.backgroundColor || 'white',
        padding: '10px',
        boxSizing: 'border-box'
    };

    if (isTouchDevice()) {
        // --- スマホ用: 選択範囲内の画像幅を基準にする ---
        var images = fragment.querySelectorAll('img');
        var canvasWidth;

        if (images.length > 0) {
            var maxWidth = 0;
            images.forEach(function(img) {
                if (img.width > maxWidth) {
                    maxWidth = img.width;
                }
            });
            // 左右のパディング(10px * 2)を加算
            canvasWidth = maxWidth + 20;
        } else {
            canvasWidth = screen.width;
        }

        snapshotContainerStyles.width = canvasWidth + 'px';
        snapshotContainerStyles.position = 'absolute';
        snapshotContainerStyles.left = '-9999px';
        snapshotContainer.appendChild(fragment);
        document.body.appendChild(snapshotContainer);
        containerToRender = snapshotContainer;
        containerToRemove = snapshotContainer;
    } else {
        // --- PC用: シンプルなコンテナ ---
        snapshotContainerStyles.width = currentRange.getBoundingClientRect().width + 'px';
        snapshotContainerStyles.position = 'absolute';
        snapshotContainerStyles.left = '-9999px';

        snapshotContainer.appendChild(fragment);
        document.body.appendChild(snapshotContainer);
        containerToRender = snapshotContainer;
        containerToRemove = snapshotContainer;
    }

    // スタイルを適用
    for (var styleKey in snapshotContainerStyles) {
        snapshotContainer.style[styleKey] = snapshotContainerStyles[styleKey];
    }

    shareButton.style.display = 'none';

    if (isTouchDevice()) {
       // --- スマホのみ: リンク付き画像をクリーンにする ---
       var linkedImages = containerToRender.querySelectorAll('a > img');
       linkedImages.forEach(function(img) {
           var link = img.parentElement;
           if (link.tagName === 'A') {
               link.parentNode.replaceChild(img, link);
           }
       });
    }

    // --- 画像生成処理 (Promiseとして定義) ---
    // Safariのセキュリティ要件を満たすため、非同期処理をPromiseとして直接渡す
    var makeImagePromise = new Promise(function(resolve, reject) {
      // --- 画像の読み込みを待つ ---
      var images = containerToRender.querySelectorAll('img');
      var imageLoadPromises = [];
      images.forEach(function(img) {
          if (!img.complete) {
              imageLoadPromises.push(new Promise(function(resolveImg) {
                  img.onload = resolveImg;
                  img.onerror = resolveImg; // エラーでも続行
              }));
          }
      });

      Promise.all(imageLoadPromises).then(function() {
          // 1. 選択範囲のHTMLをCanvasに描画
          html2canvas(containerToRender, {
              useCORS: true,
              scale: isTouchDevice() ? 1.5 : 3,
              backgroundColor: snapshotContainerStyles.backgroundColor,
          }).then(function(contentCanvas) {
          document.body.removeChild(containerToRemove);
          if (!logoUrl) return contentCanvas; // ロゴがなければそのまま返す

          // 2. ロゴ画像を読み込んで合成
          return new Promise(function(resolveLogo) {
              var img = new Image();
              img.crossOrigin = "anonymous";
              img.onload = function() {
                  var logoHeight = (isTouchDevice() ? 30 : 50) * 3;
                  var logoWidth = (img.width / img.height) * logoHeight;
                  var padding = 20 * 3;

                  // ロゴの最大幅をCanvasの80%に制限
                  var maxLogoWidth = contentCanvas.width * 0.8;
                  if (logoWidth > maxLogoWidth) {
                      logoHeight = (maxLogoWidth / logoWidth) * logoHeight;
                      logoWidth = maxLogoWidth;
                  }

                  var finalCanvas = document.createElement('canvas');
                  finalCanvas.width = contentCanvas.width;
                  finalCanvas.height = contentCanvas.height + logoHeight + padding;
                  var ctx = finalCanvas.getContext('2d');
                  ctx.fillStyle = 'white';
                  ctx.fillRect(0, 0, finalCanvas.width, finalCanvas.height);
                  ctx.drawImage(contentCanvas, 0, 0);
                  ctx.drawImage(img, (finalCanvas.width - logoWidth) / 2, contentCanvas.height + padding / 2, logoWidth, logoHeight);
                  resolveLogo(finalCanvas);
              };
              img.onerror = function() {
                  console.error('ロゴ画像の読み込みに失敗しました。');
                  resolveLogo(contentCanvas); // ロゴなしで続行
              };
              img.src = logoUrl;
          });
      }).then(function(finalCanvas) {
          // 3. 最終的なCanvasをBlobに変換
          finalCanvas.toBlob(function(blob) {
              if (blob) {
                  resolve(blob); // 成功したらBlobを返す
              } else {
                  reject(new Error('Blob generation failed')); // 失敗
              }
          }, 'image/png');
          }).catch(function(error) {
             console.error('html2canvas/toBlob failed:', error);
             reject(error);
          });
      });
    });

    // --- クリップボードへの書き込み実行 ---
    try {
      navigator.clipboard.write([
          new ClipboardItem({ 'image/png': makeImagePromise })
      ]).then(function() {
          // 成功時の処理
          alert('選択範囲の画像をクリップボードにコピーしました！\nXの投稿画面で「貼り付け」してご利用ください。');
          var text = encodeURIComponent(defaultXPostMessage + '\n\n');
          var url = encodeURIComponent(window.location.href);
          var tweetUrl = 'https://twitter.com/intent/tweet?text=' + text + '&url=' + url;
          window.open(tweetUrl, '_blank');
      }).catch(function(err) {
          // 失敗時の処理
          console.error('クリップボードへの書き込みに失敗:', err);
          alert('クリップボードへのコピーに失敗しました。お使いのブラウザが対応していない可能性があります。');
      });
    } catch (e) {
        // ClipboardItemの作成自体が失敗した場合
        console.error('ClipboardItemの作成に失敗:', e);
        alert('クリップボード機能の呼び出しに失敗しました。お使いのブラウザが対応していない可能性があります。');
    } finally {
        // 処理が完了したら選択範囲をリセット
        currentRange = null;
    }
  });
});
