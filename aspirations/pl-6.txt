                        var userScrollingUp = sectionCenter > viewportCenter;

                        if (userScrollingUp && adContainer.style.display !== 'none') {
                            adContainer.style.display = 'none';
                        }
                    }
                }
            } else {
                removeAdContainer();
            }
        }
